package "Módulo de Subastas" {

    enum EstadoActivoFisico {
        NUEVO
        USADO
        REPARADO
    }

    enum EstadoSubasta {
        ACTIVA
        FINALIZADA
        EN_REVISION
        CANCELADA
    }

    class Subasta {
        - idSubasta: String
        - titulo: String
        - descripcion: String

        + cancelarSubasta(): void
    }

    class ActivoFisico {
        - nombre: String
        - estadoFisico: EstadoActivoFisico
    }

    class Oferta {
        - fechaOferta: Date
        - estadoOferta: String
        - esGanadora: Boolean
        + agregarLinea(bien: BienConsumo, cantidad: int): void
        + eliminarLinea(bienId: String): void
        + calcularValorEstimado(): double
    }

    class LineaBienConsumo {
        - bienConsumoId: String
        - cantidad: int
        + validarCantidad(): boolean
    }

    class BienConsumo {
        - nombre: String
        - descripcion: String
        - categoria: String
    }

    class Historial {
        - ofertas: List<Oferta>
        + agregarOferta(oferta: Oferta): void
        + obtenerOferta(): List<Oferta>
    }

    Usuario "1" --> "1" Historial : tiene >
    Usuario "1" o-- "0..*" Oferta

    Oferta "1" --> "1" Imagen : posee evidencia >
