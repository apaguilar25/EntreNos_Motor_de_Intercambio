@startuml
skinparam classAttributeIconSize 0

'Historia de Usuario 4

package "Módulo de Subastas" {

    interface Persistencia {
        +cargar(): List<Object>
        +guardar(datos: List<Object>): void
    }

    class PersistenciaSubasta {
        -rutaArchivo: String = "data/subasta.JSON"
        +cargar(): List<Subasta>
        +guardar(List<Subasta>): void
    }

    class Usuario {
        -nombre: String
        -correo: String
        -sancion
          fechaSancion
          sancionActiva
    }

    note top of Usuario
        Restricción de Acceso:
        Si tiene una sanción vigente,
        crearSubasta() queda deshabilitado.
    end note

    class GestorSubasta {
        +SolicitarCrearSubasta(fechaCreacion:Date, fechaFinalizacion:Date, estado:EstadoSubasta, plazoResolucionExpirado:Boolean): void
        +cancelarSubasta(): void
        +adjudicarGanador(ofertaGanadora: Oferta): void
    }

    class Subasta {
        -idSubasta: String
        -fechaCreacion: Date
        -fechaFinalizacion: Date
        -estado: EstadoSubasta
        -plazoResolucionExpirado: Boolean
        +verificarCierreSubasta(fechaFinalizacion:Date): Boolean
        +verificarExpiracionResolucion(fechaFinalizacion: Date): Boolean
    }

    note right of Subasta
        Regla de Negocio:
        Si pasan 5 días en CERRADA_LICITACION
        sin adjudicar/cancelar:
        1. Estado pasa a CANCELADA automáticamente.
        2. Crea Sanción de 72 horas al subastador.
        3. Se notifica a todos los participantes.
    end note

    enum EstadoActivoFisico {
        NUEVO
        USADO
        REPARADO
    }

    enum EstadoSubasta {
        ACTIVA
        FINALIZADA
        EN_REVISION
        CANCELADA
    }

    class ActivoFisico {
        -nombre: String
        -estadoFisico: EstadoFisico
    }

    note right of ActivoFisico
        Restricción de Validación:
        Obligatorio: nombre, estadoFisico
        y fotografias.size() >= 1.
    end note

    class Oferta {
        -esGanadora: Boolean
    }

    class Imagen {
        -ruta: String
    }

    ' Relaciones
    PersistenciaSubasta <|.. Persistencia
    Usuario o-- Subasta : subastador
    Subasta *-- ActivoFisico
    Subasta *-- Oferta
    ActivoFisico o-- Imagen

}
@enduml
