skinparam classAttributeIconSize 0

'Historia de Usuario 3

package "Gestión_De_Transacciones_Seguras_y_Valoración_de_Confianza" {

    class GestorUsuarios {
        - usuarios: List<Usuario>
    }

    class GestorTransacciones {
        + iniciarTransaccion(demandante: Usuario, prestador: Usuario, servicio: Servicio): Transaccion
        + enviarNotificacion(destinatario: Usuario, mensaje: String, tipo: String): void
        + confirmarServicioExitoso(idTransaccion: String): void
        + retenerCreditos(): void
        + liberarCreditos(): void
        + calificarServicio(idPrestador: String, estrellas: int): void
    }

    interface Persistencia {
        + cargar(): List<Object>
        + guardar(datos: List<Object>): void
    }

    class PersistenciaUsuariosJSON {
        - RUTA_ARCHIVO: String = "data/usuarios.json"
        + cargar(): List<Usuario>
        + guardar(datos: List<Usuario>): void
    }

    class PersistenciaTransaccionesJSON {
        - RUTA_ARCHIVO: String = "data/transacciones.json"
        + cargar(): List<Transaccion>
        + guardar(datos: List<Transaccion>): void
    }

    class Servicio {
        - idServicio: String
        - descripcion: String
        - categoria: String
        - calificacion: int
    }

    class Transaccion {
        - idTransaccion: String
        - creditosRetenidos: double
        - estado: String
        + actualizarEstado(nuevoEstado: String): void
    }

    note right of Transaccion
        Segun Gemini, JSON no soporta enumerados,
        por lo que recomienda trabajar con
        estados de Strings y validarlos

        confirmarServicioExitoso esta aca porque
        Transaccion conoce Servicio y no viceversa
    end note

    class Usuario {
        - idUsuario: String
        - nombre: String
        - correoElectronico: String
        - reputacionHistorica: double
        - monedero
          creditosDisponibles: int
          creditosComprometidos: int
        + recibirNotificacion(nuevaNotificacion: Notificacion): void
        + calcularReputacionHistorica(): void
        + calcularReputacionPorServicio(servicio: Servicio): double
    }

    note left of Usuario
        calcularreputacionHistorica filtra del JSON las transacciones donde
        el usuario es prestador, agrupa por tipo de servicio,
        saca el promedio de cada uno y promedia los resultados.

        Usuario no tendra historial sino que se genera dinamicamente
        a partir de las transacciones en el JSON.
    end note

    class Notificacion {
        - idNotificacion: String
        - mensaje: String
        - tipo: String
        - estadoLectura: boolean
        + marcarNotificacionLeida(): void
    }

    note right of Notificacion
        tipo se refiere a si sera
        notificacion popup, push, por md, etc
    end note

    ' Relaciones de Realización (Interfaces)
    PersistenciaUsuariosJSON .up.|> Persistencia
    PersistenciaTransaccionesJSON .up.|> Persistencia

    ' Relaciones de Asociación Dirigida
    Transaccion "1" *-- "1" Servicio
    Transaccion "*" -- "1" Usuario : Demanda <
    Transaccion "*" -- "1" Usuario : Ofrece <
    Usuario "1" -- "*" Notificacion : posee >
}

@enduml
