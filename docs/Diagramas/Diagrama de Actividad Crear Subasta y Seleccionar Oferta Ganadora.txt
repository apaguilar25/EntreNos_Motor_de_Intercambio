@startuml

skinparam style strictuml
skinparam DefaultFontName Arial
skinparam ActivityBackgroundColor #E8F4F8
skinparam ActivityBorderColor #2C3E50
skinparam NoteBackgroundColor #FFF9C4

title Diagrama de Actividades HU4: Gestión de Subastas de Activos por Bienes de Consumo

partition "**act** Crear Subasta y Seleccionar Oferta Ganadora" {

start

:Seleccionar creación de nueva subasta;
:Desplegar formulario para creación de subasta;

fork
  :Ingresar nombre del activo;
fork again
  :Indicar estado físico del activo;
fork again
  :Cargar una o más imágenes del activo;
fork again
  :Establecer fecha de finalización de la subasta;
end fork

:Seleccionar opción para publicar la subasta;
:Mostrar resumen de la subasta y solicitar confirmación;


if (¿Qué selecciona el subastador?) then ([Confirmar y Publicar])
  :Crear la subasta con el estado "Activa";
  :Almacenar en persistencia la información de la nueva subasta;
  :Publicar la subasta;
else ([Cancelar proceso])
  :Cerrar formulario y descartar los cambios;
  stop
endif

repeat
  :Comprobar fecha actual y fecha de finalización;
repeat while (¿Se cumplió la fecha de finalización?) is ([No]) not ([Sí])

' ----- Periodo de licitación -----
  :Cambiar el estado de la subasta a "En revisión";
  :Notificar a todos los postores la finalización del periodo de licitación;
  :Notificar al subastador la finalización del periodo de licitación y
   el plazo límite de 5 días para adjudicar la subasta o cancelarla;


repeat
' ----- Plazo para que el subastador decida -----
switch (¿Qué selecciona el subastador?)

case ( [Eligió un ganador] )
  :Revelar los datos de contacto entre el ganador y el 
      subastador para realizar el intercambio;
  :Notificar los resultados a todos los participantes;
  :Cambiar el estado de la subasta a "Finalizada";
  (B)
    detach
  break


case ( \n  [Canceló la subasta] )
  :Notificar a los postores que la subasta fue
    cancelada y no se seleccionó ganador;
  :Cambiar el estado de la subasta a "Cancelada";
  (B)
    detach
  break

case (  [No ha respondido] )
  :Comparar la fecha actual con la fecha límite del plazo de resolución;

  if (¿Han pasado 5 días desde la fecha de finalización?) then ([No])
      ->;
    else ([Sí])
    
      :Notificar a los postores que no hubo ganador
           por falta de respuesta del subastador;
      :Inhabilitar los servicios de subasta de activos al subastador durante 3 días;
      :Notificar sanción al subastador;
      :Cambiar el estado de la subasta a "Cancelada";
      ->;
  endif

endswitch


repeat while (¿El plazo expiró sin recibir respuesta del subastador?) is ([No]) not ([Sí])

(B)


' ----- Unificación final -----

  : Actualizar información de
    subasta en persistencia;
    
  :Mostrar mensaje de finalización;
