@startuml

start

skinparam style strictuml
skinparam DefaultFontName Arial
skinparam ActivityBackgroundColor #E8F4F8
skinparam ActivityBorderColor #2C3E50
skinparam NoteBackgroundColor #FFF9C4

title Diagrama de Actividades HU5: Participación en Subastas de Activos

partition **act** Participar en Subasta y Realizar Seguimiento {

:Seleccionar creación de propuesta para participar en una subasta;
:Desplegar formulario de propuesta de intercambio;

fork
:Navegar por la lista de opciones de bienes de consumo;
repeat
  :Seleccionar bien de consumo;
  :Indicar cantidad a ofrecer;
repeat while (¿Agregar otro bien de consumo?) is (     [Sí]) not ( [No])

forkagain
:Cargar imagen de los bienes ofrecidos;

endfork

:Seleccionar opción para publicar la propuesta;
:Mostrar resumen de la propuesta y solicitar confirmación;

if (¿Qué selecciona el postor?) then ([Confirmar y Publicar])
  :Crear la propuesta de intercambio con el estado "Activa";
  :Almacenar en persistencia la información de la nueva propuesta;
  :Publicar la propuesta;
  :Agregar la propuesta al historial del postor;
else ([Cancelar proceso])
  :Cerrar formulario y descartar los cambios;
  stop
endif

:Mostrar mensaje de propuesta de postulación publicada exitosamente;

repeat
:Verificar notificaciones recibidas;

repeat while (¿Se ha recibido respuesta del subastador?) is (      [No]) not ([Sí])


switch (¿Cuál fue la resolución?)

case ( [El postor ganó la subasta] )
  :Recibir notificación sobre cierre de subasta
  donde ha sido seleccionado como ganador;
  :Revelar los datos de contacto entre el ganador y el
        subastador para realizar el intercambio;
  :Cambiar el estado de la propuesta a "Ganada";

case ( [El postor no ganó la subasta] )
  :Recibir notificación sobre el cierre de subasta
    donde no fue seleccionado como ganador;
  :Cambiar el estado de la propuesta a "Perdida";

case (          [El subastador canceló la subasta] )
  :Recibir notificación sobre cancelación
  de la subasta por parte del subastador;
  :Cambiar el estado de la propuesta a "Cancelada";

case (  [El plazo de selección venció sin obtener respuesta del subastador] )
  :Recibir notificación sobre cancelación de la subasta por vencimiento del
           plazo de selección sin obtener de respuesta del subastador;
  :Cambiar el estado de la propuesta a "Cancelada";

endswitch

: Actualizar información de propuesta
      de intercambio en persistencia;

:Mostrar mensaje de finalización;
