@startuml
skinparam style strictuml
skinparam DefaultFontName Arial
skinparam ActivityBackgroundColor #E8F4F8
skinparam ActivityBorderColor #2C3E50
skinparam Archivo {
    BackgroundColor White
    BorderColor Black
}

title Diagrama de Actividad HU2: Muro de Intercambio y Motor de Búsqueda

partition "**act** Navegar por el Muro de Intercambios" {
    start

    :Acceder al muro de publicaciones;

    fork
      :Mostrar saldo de créditos disponible;
    fork again
        :Desplegar lista de publicaciones con Ofertas y Solicitudes;
    endfork
    
    repeat
        
        :Navegar por la lista de publicaciones;
        
            :Seleccionar una publicación específica;
            if (¿La publicación es de un Ofertante o Demandante?) then ([Demandante])
                    fork
                        :Mostrar datos de la solicitud;
                    fork again
                        :Mostrar descripción detallada;
                    endfork
            else ([Ofertante])
                fork
                    :Mostrar datos de la oferta;
                fork again
                    :Mostrar descripción detallada;
                fork again
                    :Mostrar precio en créditos;
                fork again
                    :Mostrar promedio de calificaciones del servicio recibidas;
                endfork
            endif
        if (¿Es de su interés?) then ([Sí])
            :Seleccionar opción de enviar solicitud;
            if (¿La publicación es de un ofertante?) then ([Sí])
                if (¿Tiene créditos suficientes para adquirir el servicio?) then ([Sí])
                else ([No])
                    :Mostrar mensaje de error notificando que no tiene créditos suficientes;
                    (D)
                    note 
                    Label indicando que las acciones continuan más adelante en el diagrama
                    endnote
                    kill
                endif
            else ([No])
            endif
            :Notificar a la contraparte sobre Oportunidad de Intercambio;
            :Comparar fecha actual con la fecha de envío de solicitud;
            if (¿Han pasado 5 días de la fecha de envío?) then ([Sí])
                :Rechazar la solicitud automáticamente;
                :Notificar a ambas partes el rechazo por tiempo expirado;
            else ([No])
                if (¿Cuál fue la respuesta de la contraparte a la solicitud?) then ([Fue aceptada])
                    :Preparar inicio de transacción;
                    note right
                        Esta acción vincula este escenario (Historia 
                        de Usuario 2) con la Historia de Usuario 3.
                    endnote
                    stop
                else ([Fue rechazada])
                    :Recibir notificación de rechazo por la contraparte;
                endif
            endif
        else ([No])
            :Regresar al muro principal;
        endif
    (D)
    
    repeat while (¿Continuar navegando?) is ([Sí])
    ->[No];
    :Finalizar navegación sin selección;
    stop
}
@enduml
